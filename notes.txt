# nerdtree
o: open
cd: select as cwd
CD: change to cwd
u: up one directory
r: refresh
m: modify, add, delete file directory
I: show hidden files

# autocomplete
<C-n> :  autocomplete 
<C-x> <C-n> : only from open file
<C-x> <C-f> : autocomplete file path 

# add, sub number
<C-a> add
<C-x> subtract

# ctags
<C-]> : go to definition
g<C-]> : list all tags
<C-t> : go back up the tag stack

# tabs
:tabe
gt : tabnext
gT : tabprev


# auto complte
Ctrl-n   :      auto complet in insert mode
Ctrl-x Ctrl-n  :  only in current file
Ctrl-x Ctrl-f  : auto complet file paths


# scrolling
e : end
y : up


_g : last none space        


# mark
m<register>  : set mark
`<register>  : go to mark
